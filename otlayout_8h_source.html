126struct CoverageTable : Io::BChunk {
127 using Format = Io::BField<u16be, 0>;
128 using GlyphCount = Io::BField<u16be, 2>;
129
130 usize format() const { return get<Format>(); }
131
132 usize len() const { return get<GlyphCount>(); }
133
134 Opt<usize> coverageIndex(usize glyphId) {
135 auto s = begin().skip(4);
136
137 if (format() == 1) {
138 for (auto i : range(len())) {
139 auto glyph = s.nextU16be();
140 if (glyph == glyphId) {
141 return i;
142 }
143 }
144 }
145
146 if (format() == 2) {
147 for (auto i : range(len())) {
148 (void)i;
149 auto start = s.nextU16be();
150 auto end = s.nextU16be();
151 auto index = s.nextU16be();
152 if (start <= glyphId and glyphId <= end) {
153 return index + glyphId - start;
154 }
155 }
156 }
157
158 return NONE;
159 }

224struct GlyphPairAdjustment : LookupSubtableBase {
225 static constexpr int FORMAT = 1;
226
227 Opt<Pair<ValueRecord>> adjustments(usize prev, usize curr) {
228 auto s = begin();
229
230 // Read the table header
231 /* format = */ s.nextU16be();
232 auto coverageOffset = s.nextU16be();
233 auto valueFormat1 = s.nextU16be();
234 auto valueFormat2 = s.nextU16be();
235 auto pairSetCount = s.nextU16be();
236
237 // Lookup the coverage index for the first glyph
238 CoverageTable coverage{begin().skip(coverageOffset).remBytes()};
239 auto coverageIndex = coverage.coverageIndex(prev);
240
241 if (not coverageIndex or (*coverageIndex >= pairSetCount))
242 return NONE;
243
244 auto value1len = ValueRecord::len(valueFormat1);
245 auto value2len = ValueRecord::len(valueFormat2);
246 auto pairSetOffset = s.skip(coverageIndex.unwrap() * 2).nextU16be();
247
248 // Lookup the PairSet table for the second glyph
249 auto pairSetTable = begin().skip(pairSetOffset);
250 auto pairValueCount = pairSetTable.nextU16be();
251
252 for (usize i : range(pairValueCount)) {
253 (void)i;
254 auto secondGlyph = pairSetTable.nextU16be();
255 if (secondGlyph == curr) {
256 ValueRecord value1 = ValueRecord::read(pairSetTable, valueFormat1);
257 ValueRecord value2 = ValueRecord::read(pairSetTable, valueFormat2);
258 return Pair<ValueRecord>{value1, value2};
259 }
260
261 pairSetTable.skip(value1len + value2len);
262 }
263
264 return NONE;
265 }

269struct ClassDef : Io::BChunk {
270 Opt<usize> classOf(usize glyphId) {
271 auto s = begin();
272 auto format = s.nextU16be();
273
274 if (format == 1) {
275 auto startGlyph = s.nextU16be();
276 auto glyphCount = s.nextU16be();
277 if (startGlyph <= glyphId and glyphId < startGlyph + glyphCount) {
278 return s.skip((glyphId - startGlyph) * 2).nextU16be();
279 }
280 }
281
282 if (format == 2) {
283 auto classRangeCount = s.nextU16be();
284 for (usize i : range(classRangeCount)) {
285 (void)i;
286 auto startGlyph = s.nextU16be();
287 auto endGlyph = s.nextU16be();
288 auto glyphClass = s.nextU16be();
289 if (startGlyph <= glyphId and glyphId <= endGlyph) {
290 return glyphClass;
291 }
292 }
293 }
294
295 return NONE;
296 }

300struct ClassPairAdjustment : LookupSubtableBase {
301 static constexpr int FORMAT = 2;
302
303 Opt<Pair<ValueRecord>> adjustments(usize prev, usize curr) {
304 auto s = begin();
305
306 // Read the table header
307 /* format = */ s.nextU16be();
308 /* coverageOffset = */ s.nextU16be();
309 auto valueFormat1 = s.nextU16be();
310 auto valueFormat2 = s.nextU16be();
311 auto classDef1Offset = s.nextU16be();
312 auto classDef2Offset = s.nextU16be();
313 /* class1Count = */ s.nextU16be();
314 auto class2Count = s.nextU16be();
315
316 ClassDef prevClassDef{begin().skip(classDef1Offset).remBytes()};
317 ClassDef currClassDef{begin().skip(classDef2Offset).remBytes()};
318
319 auto prevClass = try$(prevClassDef.classOf(prev));
320 auto currClass = try$(currClassDef.classOf(curr));
321
322 auto value1len = ValueRecord::len(valueFormat1);
323 auto value2len = ValueRecord::len(valueFormat2);
324
325 auto class2Size = value1len + value2len;
326 auto class1Size = class2Count * class2Size;
327 s.skip((prevClass * class1Size) + (currClass * class2Size));
328
329 ValueRecord value1 = ValueRecord::read(s, valueFormat1);
330 ValueRecord value2 = ValueRecord::read(s, valueFormat2);
331
332 return Pair<ValueRecord>{value1, value2};
333 }
