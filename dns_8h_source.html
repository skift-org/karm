120Res<> encodeName(Io::BEmit& e, Str name) {
121 for (auto part : iterSplit(name, '.')) {

129Res<usize> decodeName(Cursor<u8> const start, Cursor<u8> curr, StringBuilder& out) {
130 usize len = 0;
131 while (not curr.ended()) {
132 auto b = curr.next();
133 len++;
134
135 if (b == 0) {
136 // End of name
137 if (out.len() > MAX_NAME_LEN)
138 return Error::invalidData("dns name too long");
139
140 return Ok(len);
141 } else if ((b & 0xc0) == 0xc0) {
142 // Pointer
143 if (curr.ended())
144 return Error::unexpectedEof("unexpected end of dns name");
145 usize off = ((b & 0x3f) << 8) | curr.next();
146 len++;
147 if (off >= start.len())
148 return Error::invalidData("invalid dns name pointer");
149 auto next = start;
150 next.next(off);
151 try$(decodeName(start, next, out));
152 return Ok(len);
153 } else {
154 // Label
155 if (b > MAX_LABEL_LEN)
156 return Error::invalidData("dns label too long");
157 if (out.len() + b + 1 > MAX_NAME_LEN)
158 return Error::invalidData("dns name too long");
159 if (out.len() > 0)
160 out.append('.');
161 out.append(curr.next(b).cast<char>());
162 len += b;
163 }
164 }
165
166 return Error::unexpectedEof("unexpected end of dns name");
167}

192 static Res<Buf<u8>> encode(Packet const& p) {
193 Io::BufferWriter buf;
194 Io::BEmit e(buf);
195
196 e.writeBytes(p.header().bytes());
197
198 for (auto& q : p._qs) {
199 try$(encodeName(e, q.name));
200 e.writeU16be(q.type);
201 e.writeU16be(q.class_);
202 }
203
204 for (auto& a : p._ans) {
205 try$(encodeName(e, a.name));
206 e.writeU16be(a.type);
207 e.writeU16be(a.class_);
208 e.writeU32be(a.ttl.toSecs());
209 e.writeU16be(a.data.len());
210 e.writeBytes(a.data);
211 }
212
213 return Ok(buf.take());
214 }
215
216 static Res<Packet> decode(Bytes buf) {
217 Io::BScan s{buf};
218
219 if (s.rem() < sizeof(Header))
220 return Error::unexpectedEof("packet too short");
221
222 Header hdr;
223 s.readTo(&hdr);
224
225 Packet pkt{hdr.id, (Flags)hdr.flags.value()};
226 Vec<Question>& qs = pkt._qs;
227 for (auto i = 0; i < hdr.qdcount; i++) {
228 StringBuilder name;
229 s.skip(try$(decodeName(buf, s.remBytes(), name)));
230
231 Question q{
232 name.take(),
233 (Type)s.nextU16be(),
234 (Class)s.nextU16be(),
235 };
236
237 qs.pushBack(std::move(q));
238 }
239
240 Vec<Answer>& ans = pkt._ans;
241 for (auto i = 0; i < hdr.ancount; i++) {
242 StringBuilder name;
243 s.skip(try$(decodeName(buf, s.remBytes(), name)));
244
245 Answer a;
246 a.name = name.take();
247 a.type = (Type)s.nextU16be();
248 a.class_ = (Class)s.nextU16be();
249 a.ttl = Duration::fromSecs(s.nextU32be());
250 auto len = s.nextU16be();
251 a.data = s.nextBytes(len);
252
253 ans.pushBack(std::move(a));
254 }
255
256 return Ok(std::move(pkt));
257 }

273 Res<Vec<Answer>> ask(Slice<Question> qs) {
274 Packet req{420, Flags::RD};
275 req._qs = qs;
276
277 logDebug("sending dns request to {}", _server);
278 try$(_conn.send(try$(Packet::encode(req)), _server));
279
280 logDebug("waiting for dns response");
281 Array<u8, 4096> buf;
282 auto [len, addr] = try$(_conn.recv(buf));
283
284 logDebug("received dns response from {}", addr);
285 if (addr != _server)
286 return Error::invalidData("received response from wrong address");
287
288 Packet resp = try$(Packet::decode(sub(buf, 0, len)));
289
290 if (resp._id != req._id)
291 return Error::invalidData("received response with wrong id");
292
293 if (resp.header().rcode() != RCode::NO_ERROR)
294 return Error::invalidData("received response with error code");
295
296 return Ok(std::move(resp._ans));
297 }

299 Res<Sys::Ip> resolve(Str host) {
300 auto qs = Vec<Question>{};
301 qs.pushBack(Question{host, Type::A, Class::IN});
302
303 auto answers = try$(ask(qs));
304 if (answers.len() == 0)
305 return Error::invalidData("no answers");
306
307 auto ans = answers[0];

309 return Error::invalidData("invalid answer type");

312 return Error::invalidData("invalid answer data");
313
314 Sys::Ip4 ip{
315 ans.data[0],
316 ans.data[1],
317 ans.data[2],
318 ans.data[3],
319 };
