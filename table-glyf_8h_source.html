82 void contourSimple(Gfx::Canvas& g, Metrics m, Io::BScan& s) const {
83 auto endPtsOfContours = s;
84 auto nPoints = s.peek(2 * (m.numContours - 1)).nextU16be() + 1u;
85 u16 instructionLength = s.skip(m.numContours * 2).nextU16be();
86
87 auto flagsScan = s.skip(instructionLength);
88
89 usize nXCoords = 0;
90 u8 flags = 0;
91 u8 flagsRepeat = 0;
92
93 for (usize i = 0; i < nPoints; i++) {
94 if (not flagsRepeat) {
95 flags = s.nextU8be();
96 if (flags & REPEAT) {
97 flagsRepeat = s.nextU8be();
98 }
99 } else {
100 flagsRepeat--;
101 }
102
103 nXCoords += flags & SimpleContour::X_SHORT_VECTOR
104 ? 1
105 : (flags & SAME_OR_POSITIVE_X ? 0 : 2);
106 }
107
108 auto xCoordsScan = s;
109 auto yCoordsScan = s.skip(nXCoords);
110
111 usize start = 0;
112 Math::Vec2f curr{};
113 flags = 0;
114 flagsRepeat = 0;
115 for (isize c = 0; c < m.numContours; c++) {
116 usize end = endPtsOfContours.nextU16be();
117
118 Math::Vec2f cp{};
119 Math::Vec2f startP{};
120 bool wasCp = false;
121
122 for (usize i = start; i <= end; i++) {
123 if (not flagsRepeat) {
124 flags = flagsScan.nextU8be();
125 if (flags & REPEAT) {
126 flagsRepeat = flagsScan.nextU8be();
127 }
128 } else {
129 flagsRepeat--;
130 }
131
132 isize x = (flags & X_SHORT_VECTOR)
133 ? ((flags & SAME_OR_POSITIVE_X) ? xCoordsScan.nextU8be() : -xCoordsScan.nextU8be())
134 : ((flags & SAME_OR_POSITIVE_X) ? 0 : xCoordsScan.nextI16be());
135
136 isize y = (flags & Y_SHORT_VECTOR)
137 ? ((flags & SAME_OR_POSITIVE_Y) ? yCoordsScan.nextU8be() : -yCoordsScan.nextU8be())
138 : ((flags & SAME_OR_POSITIVE_Y) ? 0 : yCoordsScan.nextI16be());
139
140 curr = curr + Math::Vec2f{(f64)x, (f64)-y};
141
142 if (i == start) {
143 g.moveTo(curr);
144 startP = curr;
145 } else {
146 if (flags & ON_CURVE_POINT) {
147 if (wasCp) {
148 g.quadTo(cp, curr);
149 } else {
150 g.lineTo(curr);
151 }
152 wasCp = false;
153 } else {
154 if (wasCp) {
155 auto p1 = (cp + curr) / 2;
156 g.quadTo(cp, p1);
157 cp = curr;
158 wasCp = true;
159 } else {
160 cp = curr;
161 wasCp = true;
162 }
163 }
164 }
165 }
166
167 if (wasCp) {
168 g.quadTo(cp, startP);
169 }
170
171 g.closePath();
172 start = end + 1;
173 }
174 }

189 void contourComposite(Gfx::Canvas& g, Io::BScan& s, Loca const& loca, Head const& head) const {
190 while (true) {
191 u16 flags = s.nextU16be();
192 u16 glyphIndex = s.nextU16be();
193
194 // args
195 i16 arg1 = 0, arg2 = 0;
196 if (flags & ARG_1_AND_2_ARE_WORDS) {
197 arg1 = s.nextI16be();
198 arg2 = s.nextI16be();
199 } else {
200 arg1 = (i8)s.nextU8be();
201 arg2 = (i8)s.nextU8be();
202 }
203
204 f64 tx = 0, ty = 0;
205 i16 compPoint = 0, basePoint = 0;
206
207 if (flags & ARGS_ARE_XY_VALUES) {
208 tx = arg1;
209 ty = -(f64)arg2;
210 if (flags & ROUND_XY_TO_GRID) {
211 tx = Math::round(tx);
212 ty = Math::round(ty);
213 }
214 } else {
215 compPoint = arg1;
216 basePoint = arg2;
217 logDebugIf(DEBUG_GLYF, "glyf: anchor-point composite (base={}, comp={}) not aligned", basePoint, compPoint);
218 }
219
220 f64 a = 1, b = 0, c = 0, d = 1;
221 if (flags & WE_HAVE_A_SCALE) {
222 f64 s14 = (f64)s.nextI16be() / 16384.0;
223 a = d = s14;
224 } else if (flags & WE_HAVE_AN_X_AND_Y_SCALE) {
225 a = (f64)s.nextI16be() / 16384.0;
226 d = (f64)s.nextI16be() / 16384.0;
227 } else if (flags & WE_HAVE_A_TWO_BY_TWO) {
228 a = (f64)s.nextI16be() / 16384.0;
229 b = (f64)s.nextI16be() / 16384.0;
230 c = (f64)s.nextI16be() / 16384.0;
231 d = (f64)s.nextI16be() / 16384.0;
232 }
233
234 Math::Trans2f t{a, c, b, d, tx, ty};
235
236 // resolve and render child
237 if (usize subOff = loca.glyfOffset(glyphIndex, head)) {
238 g.push();
239 g.transform(t);
240 contour(g, subOff, loca, head);
241 g.pop();
242 } else {
243 logWarn("glyf: component {} has no offset", glyphIndex);
244 }
245
246 // last component?
247 if (!(flags & MORE_COMPONENTS)) {
248 if (flags & WE_HAVE_INSTRUCTIONS) {
249 u16 ilen = s.nextU16be();
250 s.skip(ilen);
251 }
252 break;
253 }
254 }
255 }
